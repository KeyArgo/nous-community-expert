# Talio v3 — Implementation Plan

> **For Hermes:** Execute task-by-task. Frontend work — no JS test framework exists; verification = browser smoke + curl checks + the QA script from Task 0. Each task ends with a commit. Every commit auto-deploys via Gitea → GitHub → Cloudflare Pages (6h cadence already proven — last 5 runs green).

**Goal:** Evolve Talio from functional Discord archive search into a polished, shareable, viral-ready community intelligence tool.

**Architecture:** Pure static site on Cloudflare Pages. `build_index.py` generates JSON data files at deploy time. `index.html` (vanilla JS) consumes JSON client-side. No server, no API, no npm. Brand: Talio (winged sandals of Hermes).

**Tech Stack:** Python stdlib (build scripts), Vanilla JS + Canvas (frontend), Cloudflare Pages (hosting), GitHub Actions (every 6h cron → CF deploy hook), Gitea (primary remote) → GitHub mirror.

**Huddle constraints (deepseekv4flash + minimaxm3 skeptic):**
- Search is the hero — single-column reading flow for results
- Virality from utility (shareable URLs, OG unfurl previews) > badges
- Gamification is hollow on static data — medals are rank-based, awards only go UP
- Visual polish matters: winged sandal logo, embedded graphs, dark theme
- For a static site, OG image must be static fallback (no per-page dynamic)

**Code audit findings (this is v3, not v2 — fixes applied):**
- `_data.daily_counts` doesn't exist → must be computed in build_index.py
- `max_streak`, `msg_per_chunk`, `dates_active` not in `all_author_stats` → must be added
- Profile card renders badges in separate path from leaderboard → centralize in `renderBadges()` helper
- Function is `load()` not `loadData()`; helper is `esc()` not `escHtml()`; var is `etxt` not `displayText`
- OG meta tags missing entirely (biggest leverage gap for virality)
- No first-time-visitor intro/empty state (shared URLs bounce without context)

**Measured constraints (verified against the live data):**

```python
#!/usr/bin/env python3
"""Validate generated JSON has expected fields. Run after every build_index.py run."""
import json
import sys
from pathlib import Path

BASE = Path(__file__).resolve().parent
errors = []

def check(cond, msg):
    if not cond:
        errors.append(f"  ✗ {msg}")
    else:
        print(f"  ✓ {msg}")

try:
    data = json.load(open(BASE / "search-data.json"))
    idx = json.load(open(BASE / "search-index.json"))
except Exception as e:
    print(f"FATAL: cannot load JSON: {e}")
    sys.exit(1)

print("== search-data.json ==")
check("last_updated" in data, "has last_updated")
check("chunks" in data and len(data["chunks"]) > 0, f"has chunks ({len(data.get('chunks',[]))})")
check("stats" in data, "has stats")
check("stats" in data and "channels" in data["stats"], "stats has channels")
check("sample_queries" in data and len(data["sample_queries"]) > 0, f"has sample_queries ({len(data.get('sample_queries',[]))})")
check("leaderboard" in data and len(data["leaderboard"]) > 0, f"has leaderboard ({len(data.get('leaderboard',[]))})")
check("streaks" in data, "has streaks")
check("top_days" in data, "has top_days")

# Field-level checks (gated by feature flag presence)
if data.get("leaderboard"):
    first = data["leaderboard"][0]
    check("name" in first, "leaderboard[0] has name")

# T6: daily_counts (added in Task 6a)
if "daily_counts" in data.get("stats", {}):
    dc = data["stats"]["daily_counts"]
    check(isinstance(dc, dict) and len(dc) > 0, f"daily_counts present ({len(dc)} days)")

# T3: medal + awards (added in Task 3)
if data.get("leaderboard"):
    first = data["leaderboard"][0]
    if "medal" in first:
        check(isinstance(first["medal"], (list, tuple)) and len(first["medal"]) == 2, "leaderboard[0].medal is [id, name]")
    if "awards" in first:
        check(isinstance(first["awards"], list), "leaderboard[0].awards is list")
        check(len(first["awards"]) <= 2, f"awards cap ok ({len(first['awards'])} ≤ 2)")

print("== search-index.json ==")
check("terms" in idx and len(idx["terms"]) > 0, f"has terms ({len(idx.get('terms',{}))})")
check("doc_count" in idx, "has doc_count")
check("avg_dl" in idx, "has avg_dl")

if errors:
    print(f"\nFAILED ({len(errors)} errors):")
    for e in errors:
        print(e)
    sys.exit(1)
print("\nALL OK")
```

```python
def extract_trending(chunks, count=12):
    """Extract trending terms from recent chunks using simple TF."""
    from collections import Counter
    import re

    if not chunks:
        return SAMPLE_QUERIES

    now = datetime.utcnow()
    recent_texts = []
    for c in chunks:
        et = c.get("end_time", "") or c.get("start_time", "")
        if et:
            try:
                d = datetime.fromisoformat(et.replace("Z", "+00:00"))
                if (now - d).days <= 7:
                    recent_texts.append(c.get("text", ""))
            except (ValueError, TypeError):
                recent_texts.append(c.get("text", ""))

    if len(recent_texts) < 50:
        sorted_chunks = sorted(
            chunks,
            key=lambda c: c.get("end_time", "") or c.get("start_time", "") or "",
            reverse=True
        )
        recent_texts = [c.get("text", "") for c in sorted_chunks[:200]]

    stopwords = {
        "the","and","for","with","that","this","from","have","will","just",
        "about","what","when","they","your","you","are","can","not","but","all","any",
        "our","has","had","was","were","been","into","out","like","make","get","use",
        "work","know","think","take","come","way","good","more","some","time","very",
        "now","than","then","only","also","could","should","would","thats","dont",
        "im","ive","hes","shes","lets","https","http","www","com","bot","user",
        "msg","id","num","val","etc","api","url","json"
    }

    word_counts = Counter()
    for text in recent_texts:
        words = [
            w for w in re.split(r"[^a-z0-9]+", text.lower())
            if len(w) > 3 and w not in stopwords and not w.isdigit()
        ]
        for w in words:
            word_counts[w] += 1

    top_words = [w for w, _ in word_counts.most_common(30)]
    phrases = []
    for text in recent_texts:
        tl = text.lower()
        for i in range(len(top_words) - 1):
            phrase = f"{top_words[i]} {top_words[i+1]}"
            if phrase in tl and phrase not in phrases:
                phrases.append(phrase)
                if len(phrases) >= count:
                    break
        if len(phrases) >= count:
            break

    return phrases[:count] if len(phrases) >= count else SAMPLE_QUERIES[:count]
```

```python
def assign_medal(rank, name):
    """One medal per user based on rank. Founder gets Caduceus."""
    if name == "teknium":
        return ["caduceus", "Founder"]
    if rank is None:
        return ["ribbon", "Ribbon"]
    if rank <= 10:
        return ["gold", "Gold"]
    if rank <= 25:
        return ["silver", "Silver"]
    if rank <= 50:
        return ["bronze", "Bronze"]
    return ["ribbon", "Ribbon"]


def assign_awards(user_stats):
    """Stable contribution awards from lifetime stats. Only go up."""
    awards = []
    channels = len(user_stats.get("channels", set()) or set())
    first_seen = user_stats.get("first_seen", "") or ""
    max_streak = user_stats.get("max_streak", 0) or 0
    dates_active = user_stats.get("dates_active", set()) or set()
    msg_per_chunk = user_stats.get("msg_per_chunk", 0) or 0

    if channels >= 3:
        awards.append(["explorer", "Explorer"])
    if first_seen and first_seen <= "2025-01-01":
        awards.append(["pioneer", "Pioneer"])
    if max_streak >= 30:
        awards.append(["streak", "Streak"])
    if len(dates_active) >= 50:
        awards.append(["pillar", "Pillar"])
    if msg_per_chunk > 15:
        awards.append(["sage", "Sage"])
    if channels >= 4:
        awards.append(["diplomat", "Diplomat"])

    return awards[:2]  # Cap at 2 per user
```

```js
function highlightText(text, query) {
  if (!query || !text) return esc(text);
  const escaped = esc(text);
  const terms = query.toLowerCase().split(/[^a-z0-9]+/).filter(t => t.length > 1);
  if (!terms.length) return escaped;
  let result = escaped;
  for (const term of terms) {
    const safe = term.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
    const re = new RegExp(`(${safe})`, 'gi');
    result = result.replace(re, '<mark style="background:rgba(0,163,212,0.3);color:var(--primary-light);border-radius:2px;padding:0 2px">$1</mark>');
  }
  return result;
}
```

```js
function drawLine(canvas, points, color) {
  if (!canvas) return;
  const ctx = canvas.getContext('2d');
  const w = canvas.clientWidth, h = canvas.clientHeight;
  canvas.width = w * 2; canvas.height = h * 2;
  ctx.scale(2, 2);
  ctx.clearRect(0, 0, w, h);
  if (!points.length) return;
  const stepX = w / Math.max(points.length - 1, 1);
  const maxV = Math.max(...points) || 1;
  ctx.strokeStyle = color;
  ctx.lineWidth = 1.5;
  ctx.beginPath();
  points.forEach((v, i) => {
    const x = i * stepX;
    const y = h - (v / maxV) * (h - 10);
    if (i === 0) ctx.moveTo(x, y); else ctx.lineTo(x, y);
  });
  ctx.stroke();
}

function drawBars(canvas, values, color) {
  if (!canvas) return;
  const ctx = canvas.getContext('2d');
  const w = canvas.clientWidth, h = canvas.clientHeight;
  canvas.width = w * 2; canvas.height = h * 2;
  ctx.scale(2, 2);
  ctx.clearRect(0, 0, w, h);
  if (!values.length) return;
  const maxV = Math.max(...values) || 1;
  const slot = w / values.length;
  const bw = slot * 0.7;
  const gap = slot * 0.3;
  values.forEach((v, i) => {
    const x = i * slot + gap / 2;
    const bh = (v / maxV) * (h - 10);
    ctx.fillStyle = color;
    ctx.fillRect(x, h - bh, bw, bh);
  });
}
```

## Task 10: Service Worker for Instant Cache

**Objective:** Cache JSON + static assets so returning visitors load instantly.

**Files:**
- Create: `sw.js`
- Modify: `index.html` (register)

**Step 1: Create sw.js**

```js
const CACHE = 'talio-v2';
const ASSETS = [
  '/',
  '/index.html',
  '/search-data.json',
  '/search-index.json',
  '/metadata.json',
  '/logo.svg',
  '/favicon.svg',
  '/og-image.svg',
];

self.addEventListener('install', e => {
  e.waitUntil(caches.open(CACHE).then(cache => cache.addAll(ASSETS)));
  self.skipWaiting();
});

self.addEventListener('activate', e => {
  e.waitUntil(
    caches.keys().then(keys => Promise.all(
      keys.filter(k => k !== CACHE).map(k => caches.delete(k))
    ))
  );
  self.clients.claim();
});

self.addEventListener('fetch', e => {
  e.respondWith(
    caches.match(e.request).then(cached => {
      const fetchPromise = fetch(e.request).then(response => {
        if (response.ok && e.request.method === 'GET') {
          const clone = response.clone();
          caches.open(CACHE).then(cache => cache.put(e.request, clone));
        }
        return response;
      }).catch(() => cached);
      return cached || fetchPromise;
    })
  );
});
```

**Step 2: Register in index.html (insert before `load();` at line 478)**

```js
if ('serviceWorker' in navigator) {
  window.addEventListener('load', () => {
    navigator.serviceWorker.register('/sw.js').catch(() => {});
  });
}
```
